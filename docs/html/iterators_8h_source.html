66// TRICK : The || ({ x.destroy(&x); 0; }) executes the destroy function of the iterator when it ends
67// The destroy function is only called when the previous condition is false, and then evaluates to 0 (false)
68// This uses the GNU extension of "Statement Expressions"
69#define FOR_EACH(type_iterated, iterated, iterator, end_cond) \
70 for (type_iterated iterated = (iterator).next(&(iterator)); (end_cond) || ({ \
71 (iterator).destroy(&(iterator)); \
72 0; \
73 }); \
74 (iterated) = (iterator).next(&(iterator)))
