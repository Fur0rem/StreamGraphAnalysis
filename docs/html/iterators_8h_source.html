93// TRICK : The || ({ x.destroy(&x); 0; }) executes the destroy function of the iterator when it ends
94// The destroy function is only called when the previous condition is false, and then evaluates to 0 (false)
95// This uses the GNU extension of "Statement Expressions"
96#define SGA_FOR_EACH(type_iterated, iterated, iterator, end_cond) \
97 for (type_iterated iterated = (iterator).next(&(iterator)); (end_cond) || ({ \
98 (iterator).destroy(&(iterator)); \
99 0; \
100 }); \
101 (iterated) = (iterator).next(&(iterator)))
