33// The || ({ x.destroy(&x); 0; }) is a trick to execute the destroy function of the iterator when it ends
34// The destroy function is only called when the previous condition is false, and then evaluates to 0 (false)
35// This uses the GNU extension of "Statement Expressions"
36#define FOR_EACH(type_iterated, iterated, iterator, end_cond) \
37 for (type_iterated iterated = (iterator).next(&(iterator)); (end_cond) || ({ \
38 (iterator).destroy(&(iterator)); \
39 0; \
40 }); \
41 (iterated) = (iterator).next(&(iterator)))
